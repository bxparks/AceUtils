88  static uint8_t tokenize(char* line, const char** argv, uint8_t argvSize) {
89  char* token = strtok(line, DELIMS);
90  int argc = 0;
91  while (token != nullptr && argc < argvSize) {
92  argv[argc] = token;
93  argc++;
94  token = strtok(nullptr, DELIMS);
