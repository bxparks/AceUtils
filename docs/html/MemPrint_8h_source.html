58 template<typename TYPE>
59 class MemPrintBase: public Print {
60  public:
61  size_t write(uint8_t c) override {
62  if (index_ < size_ - 1) {
63  buf_[index_] = c;
64  index_++;
65  return 1;
66  } else {
67  return 0;
68  }
69  }
70 
71  size_t write(const uint8_t *buf, size_t size) override {
72  if (buf == nullptr) return 0;
73 
74  size_t n = 0;
75  while (size-- > 0) {
76  size_t ret = write(*buf++);
77  if (ret == 0) break;
78  n++;
79  }
80  return n;
81  }

83 // ESP32 version of Print class does not define a virtual flush() method so
84 // we can't use the 'override' keyword.
85 #ifdef ESP32
86  void flush() {
87  index_ = 0;
88  }
89 #else
90  void flush() override {
91  index_ = 0;
92  }

100  const char* getCstr() const {
101  buf_[index_] = '\0';
102  return buf_;
103  }

105  protected:
106  MemPrintBase(TYPE size, char* buf):
107  size_(size),
108  buf_(buf) {}
109 
110  private:
111  TYPE const size_;
112  TYPE index_ = 0;
113 
114  // This is the pointer to actualBuf_ which is defined in the subclasses.
115  // Instead of storing it (and taking up precious RAM), maybe there's a way
116  // to calculate this pointer by simply extending the pointer from the last
117  // element of this object (i.e. &index_), and thereby saving some memory.
118  // But I am not convinced that I have the knowledge do that properly
119  // without triggering UB (undefined behavior) in the C++ language. Do I
120  // define buf_[0] here and will it point exactly where actualBuf_[] is
121  // allocated? Or do I use [&MemPrintBase + sizeof(MemPrintBase)] to
122  // calculate the pointer to actualBuf_. I just don't know what's actually
123  // allowed in the language spec versus something that works by pure luck on
124  // a particular microcontroller and compiler. So I'll pay the cost of the 2
125  // extra bytes (8-bit) or 4 extra bytes (32-bit processors) of RAM and
126  // store the pointer to actualBuf_ explicitly here in the base class.
127  char* const buf_;
128 };
